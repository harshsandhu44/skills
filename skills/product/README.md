  mkdir -p .claude/skills/write-user-stories
  curl -o .claude/skills/write-user-stories/SKILL.md \
    https://raw.githubusercontent.com/harshsandhu44/skills/main/skills/product/write-user-stories/SKILL.md

  Replace write-user-stories with any skill name above.
